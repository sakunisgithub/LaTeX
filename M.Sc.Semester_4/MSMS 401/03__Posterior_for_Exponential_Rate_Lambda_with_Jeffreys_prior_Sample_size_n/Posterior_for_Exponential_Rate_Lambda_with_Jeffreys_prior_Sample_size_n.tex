\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Problem}}

Consider a random sample of size $n$ from $\text{Exp}(\lambda)$, $\lambda$ is the rate parameter. Take a non-informative prior for $\lambda$. Obtain the posterior distribution of $\lambda$, hence obtain Bayes estimator of $\lambda$ assuming squared-error loss function.

\section*{\faArrowAltCircleRight[regular] \textcolor{blue}{Solution}}

Let $x_1, x_2, \ldots, x_n \overset{iid}{\sim} f(x \mid \lambda)$ where

\begin{align*}
f(x \mid \lambda) = \lambda e^{-\lambda x}.
\end{align*}

Jeffreys prior for $\lambda$ will be $\lambda \sim \pi(\lambda) = \sqrt{I(\lambda)}$. Now, 

\begin{align*}
f(x \mid \lambda) &= \lambda e^{-\lambda x} \\[1em]
\Rightarrow \ln f(x \mid \lambda) &= \ln \lambda-\lambda x \\[1em]
\Rightarrow \dfrac{\partial}{\partial \lambda} \ln f(x\mid \lambda) &=
\dfrac{1}{\lambda} - x \\[1em]
\Rightarrow \dfrac{\partial^2}{\partial \lambda^2} \ln f(x\mid \lambda) &= -\dfrac{1}{\lambda^2}.
\end{align*}

\begin{align*}
\therefore I(\lambda) &= -E\left[ \dfrac{\partial^2}{\partial \lambda^2} \ln f(X \mid \lambda) \right] \\[1em]
&= \dfrac{1}{\lambda^2}
\end{align*}

\begin{align*}
\therefore \pi(\lambda) = \sqrt{\dfrac{1}{\lambda^2}} = \dfrac{1}{\lambda}, \; \lambda > 0.
\end{align*}

\newpage

Hence our non-informative prior for $\lambda$ is

\begin{align*}
\pi(\lambda) = \dfrac{1}{\lambda}.
\end{align*}

The likelihood for $\lambda$ is

\begin{align*}
f(\mathbf{x} \mid \lambda) &= \prod \limits_{i = 1}^{n} \lambda e^{-\lambda x_i} \\[1em]
&= \lambda^n \cdot \exp \left\{ - \lambda \sum \limits_{i = 1}^{n} x_i \right\}
\end{align*}

We compute the posterior distribution of $\lambda$ as
\begin{align*}
p(\lambda \mid \mathbf{x}) &= \dfrac{f(\mathbf{x} \mid \lambda) \, \cdot \, \pi(\lambda)}{\displaystyle \int\limits_{0}^{\infty} f(\mathbf{x} \mid \lambda) \, \cdot \, \pi(\lambda) \, d\lambda} \\[1em]
&= \dfrac{\dfrac{1}{\lambda} \cdot \lambda^n \cdot \exp \left\{ - \lambda \sum \limits_{i = 1}^{n} x_i \right\}}{\displaystyle{\int \limits_{0}^{\infty} \dfrac{1}{\lambda} \cdot \lambda^n \cdot \exp \left\{ - \lambda \sum \limits_{i = 1}^{n} x_i \right\}} \, d\lambda} \\[1em]
&= \dfrac{\lambda^{n-1} \cdot \exp \left\{ - \lambda \sum \limits_{i = 1}^{n} x_i \right\}}{\Gamma(n) \Big/ \left( \sum
 \limits_{i = 1}^{n} x_i \right)^n} \\[1em]
&= \dfrac{\left( \sum \limits_{i = 1}^{n} x_i \right)^n}{\Gamma(n)} \cdot \lambda^{n-1} \cdot \exp \left\{ - \lambda \sum \limits_{i = 1}^{n} x_i \right\}.
\end{align*}

which we identify as the PDF of Gamma$\left(\text{shape} = n, \text{rate} = \sum \limits_{i = 1}^{n} x_i\right)$.

\vspace{0.5cm}

$\therefore$ \textit{a posteriori} $\lambda \mid \mathbf{x} \sim \text{Gamma}\left(n, \sum \limits_{i = 1}^{n} x_i \right)$.

\vspace{0.5cm}

{\fontsize{12}{20} \gloriahallelujahfont Under squared error loss function, Bayes estimate is the posterior mean.}

\vspace{0.5cm}

Thus,

\begin{align*}
\hat{\lambda}_{Bayes} = \dfrac{n}{\sum \limits_{i = 1}^{n} x_i} = \dfrac{1}{\bar{x}}.
\end{align*}
